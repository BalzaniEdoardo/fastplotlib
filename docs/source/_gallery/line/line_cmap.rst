    xs = np.linspace(-10, 10, 100)
    # sine wave
    ys = np.sin(xs)

    # cosine wave
    ys = np.cos(xs) - 5
    cosine = np.dstack([xs, ys])[0]

    # cmap_values from an array, so the colors on the sine line will be based on the sine y-values
